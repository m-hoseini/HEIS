library(tidyverse)

list_HH <- list.files(path = "./../exported", pattern = "^HH.*\\Rds$")

HH <- NULL
for (year in list_HH) {
  HH <- assign(year, readRDS(paste0("./../exported/",year))) %>% 
    filter(!is.na(weight)) %>%
    mutate(year = parse_number(year)) %>%
    bind_rows(HH)
  rm(list=year)
}
